## 1. Charger facilités OSM + départements GADM

```{webr}
#| autorun: true
# sf = manipulation de données spatiales (points, polygones).
# ggplot2 = graphiques. dplyr = transformations de tableaux.
library(sf)
library(ggplot2)
library(dplyr)

# OpenStreetMap (OSM) = base mondiale collaborative.
# On lit les facilités de santé extraites via l'API Overpass (requête OSM).
# GADM = polygones administratifs ; ADM2 = niveau département.
etabs <- read.csv("_commons/data/cmr_sante/etablissements_sante_osm.csv")

# st_make_valid() repare les geometries invalides (vertices dupliques,
# auto-intersections). GADM ADM2 du Cameroun en contient quelques-unes
# qui font planter st_join avec "Edge 0 is degenerate". A appliquer
# systematiquement apres read_sf() de couches publiques imparfaites.
adm2 <- read_sf("_commons/data/gadm41_CMR_2.json") |> st_make_valid()

n <- nrow(etabs)

# Combien de points OSM, combien de départements, et répartition par type.
cat("Facilités OSM   :", n, "\n")
cat("Départements    :", nrow(adm2), "\n")
print(table(etabs$categorie))
```

## 3. Ratio R — motifs ponctuels (sur les points)

```{webr}
# Matrice des coordonnées des n facilités.
coords <- cbind(etabs$longitude, etabs$latitude)

# Pour chaque point, on cherche la distance au voisin le plus proche.
# Boucle simple, sans librairie : portable WebR.
dist_nn <- numeric(n)
for (i in seq_len(n)) {
  d <- sqrt((coords[i,1] - coords[,1])^2 + (coords[i,2] - coords[,2])^2)
  d[i] <- Inf            # on s'exclut soi-même
  dist_nn[i] <- min(d)
}

# Comparaison entre la distance moyenne observée et la distance attendue
# si les points étaient distribués au hasard (CSR) dans l'emprise.
D_obs <- mean(dist_nn)
aire  <- diff(range(etabs$longitude)) * diff(range(etabs$latitude))
D_att <- 0.5 / sqrt(n / aire)
R     <- D_obs / D_att     # R < 1 : regroupement ; R > 1 : dispersion

cat("D observé :", round(D_obs, 5), "\n")
cat("D attendu :", round(D_att, 5), "\n")
cat("Ratio R   :", round(R, 4), "\n")
cat("Verdict   :", ifelse(R < 0.85, "REGROUPÉ",
                   ifelse(R > 1.15, "DISPERSÉ", "ALÉATOIRE")), "\n")
```

## 4. KDE — densité par noyau (sur les points)

```{webr}
# KDE (Kernel Density Estimation) = lissage : on remplace chaque point
# par une petite "cloche" gaussienne et on additionne.
# Le bandwidth h règle la largeur des cloches (règle de Scott).
h_scott <- 1.06 * sd(etabs$longitude) * n^(-1/5)

# Grille régulière sur l'emprise du Cameroun (80 x 80 cellules).
n_grille <- 80
lon_seq <- seq(min(etabs$longitude) - 0.5, max(etabs$longitude) + 0.5, length.out = n_grille)
lat_seq <- seq(min(etabs$latitude)  - 0.5, max(etabs$latitude)  + 0.5, length.out = n_grille)
grille  <- expand.grid(lon = lon_seq, lat = lat_seq)

# Pour chaque cellule de grille, on additionne les contributions de tous les points.
kde <- numeric(nrow(grille))
for (i in seq_len(nrow(grille))) {
  d2 <- (grille$lon[i] - etabs$longitude)^2 + (grille$lat[i] - etabs$latitude)^2
  kde[i] <- sum(exp(-d2 / (2 * h_scott^2)))
}
grille$densite <- kde / (n * 2 * pi * h_scott^2)

# Carte de densité : palette séquentielle (clair = faible, foncé = fort).
ggplot() +
  geom_raster(data = grille, aes(lon, lat, fill = densite), interpolate = TRUE) +
  geom_sf(data = adm2, fill = NA, color = "grey50", linewidth = 0.12) +
  scale_fill_gradientn(name = "Densité KDE\n(facilités / deg²)",
                       colors = c("#FFFFF0","#FFEDA0","#FD8D3C","#E31A1C","#800026")) +
  geom_point(data = etabs, aes(longitude, latitude),
             color = "white", size = 0.4, alpha = 0.6) +
  labs(title = "Densité par noyau (KDE) des facilités de santé OSM",
       subtitle = paste0("Méthode : Kernel Density Estimation · h_Scott = ",
                         round(h_scott, 3), " ° (~", round(h_scott * 111, 0), " km)"),
       caption = "Source : OpenStreetMap Overpass · GADM v4.1 · IFORD x GDSG 2026",
       x = NULL, y = NULL) +
  theme_minimal() +
  theme(plot.title = element_text(face = "bold"))
```

## 5. Agrégation par département ADM2 (jointure spatiale)

C'est ici qu'on construit la **variable réelle** d'analyse statistique : `n_facilites` par département.

```{webr}
library(tidyr)

# Agrégation spatiale = on assigne chaque point au polygone qui le contient,
# puis on compte. C'est la base pour passer des "points" aux "zones".
etabs_in_dep <- points_sf |>
  st_join(adm2 |> select(GID_2, NAME_2, NAME_1), join = st_within)

# On compte les facilités par identifiant de département (GID_2).
n_par_dep <- etabs_in_dep |>
  st_drop_geometry() |>
  filter(!is.na(GID_2)) |>
  count(GID_2, name = "n_facilites")

# On recolle les comptages à la couche des polygones (NA -> 0).
adm2_n <- adm2 |>
  left_join(n_par_dep, by = "GID_2") |>
  mutate(n_facilites = tidyr::replace_na(n_facilites, 0L))

# Top 10 départements les plus équipés
adm2_n |>
  st_drop_geometry() |>
  arrange(desc(n_facilites)) |>
  select(NAME_1, NAME_2, n_facilites) |>
  head(10)
```

## 7. Moran's I global manuel (sur le comptage par département)

Matrice de voisinage entre départements = **contiguïté** (deux départements voisins s'ils partagent une frontière). On la calcule à la main avec `st_touches()`.

```{webr}
# On reprojette en mètres (EPSG:3119, Lambert Cameroun) pour calculs géométriques fiables.
adm2_n_lambert <- adm2_n |>
  st_transform(3119) |>
  st_make_valid() |>
  filter(!st_is_empty(geometry))

# Matrice de voisinage W : W[i,j] = 1/k si j est voisin de i (k voisins), 0 sinon.
# st_touches() = renvoie pour chaque polygone la liste de ceux qui le touchent.
m <- nrow(adm2_n_lambert)
touches <- st_touches(adm2_n_lambert)

W <- matrix(0, m, m)
for (i in seq_len(m)) {
  v <- touches[[i]]
  if (length(v) > 0) W[i, v] <- 1 / length(v)  # normalisation par ligne (style "W")
}

# Moran's I = indicateur global d'autocorrélation spatiale.
# Mesure si les valeurs proches géographiquement se ressemblent.
x  <- adm2_n_lambert$n_facilites
xc <- x - mean(x)              # écarts à la moyenne
S0 <- sum(W)
num <- sum(W * outer(xc, xc))  # produits des écarts pondérés par W
den <- sum(xc^2)
I_moran <- (m / S0) * (num / den)

cat("Nombre de départements :", m, "\n")
cat("Moran's I (contiguïté) :", round(I_moran, 4), "\n")
cat("Lecture : I > 0 = clusters voisins similaires | I < 0 = damier | I = 0 = aléatoire\n")
```

## 8. Test de significativité par permutation

```{webr}
# Test par permutation Monte Carlo : on mélange aléatoirement les valeurs
# entre départements 499 fois et on recalcule I. Si le I observé est rare
# dans cette distribution, on conclut à une vraie structure spatiale.
# C'est l'alternative aux tests asymptotiques de spdep (non portable WebR).
n_perm <- 499
I_perm <- numeric(n_perm)
for (p in seq_len(n_perm)) {
  x_perm <- sample(x)            # mélange = on casse la géographie
  xc_p <- x_perm - mean(x_perm)
  I_perm[p] <- (m / S0) * (sum(W * outer(xc_p, xc_p)) / sum(xc_p^2))
}

# p-valeur = proportion de permutations qui donnent un I au moins aussi extrême.
p_value <- (sum(I_perm >= I_moran) + 1) / (n_perm + 1)

cat("I observé      :", round(I_moran, 4), "\n")
cat("Moyenne perm   :", round(mean(I_perm), 4), "\n")
cat("p-valeur (499) :", round(p_value, 4), "\n")
cat("Verdict        :",
    ifelse(p_value < 0.05, "AUTOCORRÉLATION SIGNIFICATIVE",
                           "Pas de structure détectable"), "\n")
```

## 9. LISA local par département

```{webr}
# LISA (Local Indicators of Spatial Association) = on calcule I localement
# pour chaque département. Permet d'identifier où sont les clusters.
# z_v = valeur centrée-réduite du département.
# z_lag = moyenne pondérée des voisins.
z_v   <- as.numeric(scale(x))
z_lag <- as.numeric(W %*% z_v)
I_local <- z_v * z_lag

# Test local par permutation (99 mélanges pour chaque département).
n_perm_loc <- 99
p_local <- numeric(m)
for (i in seq_len(m)) {
  autres <- setdiff(seq_len(m), i)
  Ip <- numeric(n_perm_loc)
  for (p in seq_len(n_perm_loc)) {
    zf <- z_v
    zf[autres] <- sample(z_v[autres])
    lag_i <- sum(W[i, ] * zf)
    Ip[p] <- z_v[i] * lag_i
  }
  p_local[i] <- (sum(abs(Ip) >= abs(I_local[i])) + 1) / (n_perm_loc + 1)
}

# Classement en 4 quadrants + non significatif :
# HH = département élevé entouré d'élevés (cluster équipé)
# LL = département faible entouré de faibles (désert médical)
# HL = élevé isolé entre faibles (îlot)
# LH = faible isolé entre élevés (carence locale)
adm2_n_lambert$quadrant <- ifelse(
  z_v > 0 & z_lag > 0 & p_local < 0.05, "HH – Cluster équipé",
  ifelse(z_v < 0 & z_lag < 0 & p_local < 0.05, "LL – Désert médical",
  ifelse(z_v > 0 & z_lag < 0 & p_local < 0.05, "HL – Ilot équipé",
  ifelse(z_v < 0 & z_lag > 0 & p_local < 0.05, "LH – Carence locale",
                                                "Non significatif"))))

print(table(adm2_n_lambert$quadrant))
```
